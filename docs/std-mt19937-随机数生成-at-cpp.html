std::mt19937 c++ 随机数

立即使用： 

#include <random>
#include <iostream>

int main()
{
	std::random_device rd;
	std::mt19937 gen{rd()};
	int x = gen();	// x 就是生成的随机数了
	std::cout << "随机数：" << x << std::endl;
}
